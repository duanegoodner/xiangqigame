27class PlayerCommandInterpreter:
28 """
29 Converts command line input related to a player into PlayerInput object.
30 """
31 _player_input_dispatch = {
32 "ai": PlayerType.AI,
33 "person": PlayerType.HUMAN,
34 None: PlayerType.AI,
35 }
36
37 _minimax_key_size_dispatch = {
38 64: MinimaxMoveEvaluator64,
39 128: MinimaxMoveEvaluator128,
40 }
41
42 _default_key_size = 64
43
44 _default_strength = 4
45
46 def __init__(
47 self,
48 player_input: str,
49 algo_input: str,
50 strength_input: int,
51 key_size_input: int,
52 ):
53 self.player_input = player_input
54 self.algo_input = algo_input
55 self.strength_input = strength_input
56 self.key_size_input = key_size_input
57
58 def _get_key_size(self) -> int:
59 if self.key_size_input is None:
60 return self._default_key_size
61 else:
62 return self.key_size_input
63
64 def _get_strength(self) -> int:
65 if self.strength_input is None:
66 return self._default_strength
67 else:
68 return self.strength_input
69
70 def interpret_command(self) -> PlayerInput:
71 player_type = self._player_input_dispatch[self.player_input]
72 if player_type == PlayerType.HUMAN:
73 algo = strength = key_size = None
74 else:
75 if self.algo_input == "random":
76 algo = EvaluatorType.RANDOM
77 strength = key_size = None
78 else:
79 key_size = self._get_key_size()
80 algo = EvaluatorType.MINIMAX
81 strength = self._get_strength()
82 return PlayerInput(
83 player_type=player_type,
84 algo=algo,
85 strength=strength,
86 key_size=key_size,
87 )
