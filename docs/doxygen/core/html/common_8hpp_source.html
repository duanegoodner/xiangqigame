115inline bool operator==(
116 const PiecePointsArray_t &a,
117 const PiecePointsArray_t &b
118) {
119 bool are_equal = true;
120
121 for (auto rank = 0; rank < gameboard::kNumRanks; rank++) {
122 for (auto file = 0; file < gameboard::kNumFiles; file++) {
123 if (a[rank][file] != b[rank][file]) {
124 are_equal = false;
125 return are_equal;
126 }
127 }
128 }
129 return are_equal;
130}
