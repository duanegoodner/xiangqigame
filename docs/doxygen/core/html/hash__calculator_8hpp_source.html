125 void ImplementFullBoardStateCalc(const BoardMap_t &board_map) {
126 board_state_ = 0;
127 for (size_t rank = 0; rank < kNumRanks; rank++) {
128 for (size_t file = 0; file < kNumFiles; file++) {
129 if (board_map[rank][file].piece_color != 0) {
130 board_state_ = board_state_ ^ zkeys_.GetHashValue(
131 board_map[rank][file].piece_color,
132 board_map[rank][file].piece_type,
133 BoardSpace{(int)rank, (int)file}
134 );
135 }
136 }
137 }
138 };
139 void ImplementRecordTrData(
140 int search_depth,
141 MinimaxResultType result_type,
142 BestMoves &best_moves
143 ) {
144 transposition_table_.RecordData(board_state_, search_depth, result_type, best_moves);
145 }
146
147 TranspositionTableSearchResult ImplementGetTrData(int search_depth) {
148 return transposition_table_.GetData(board_state_, search_depth);
149 }
150
151private:
152 ZobristKeys<KeyType> zkeys_;
153 KeyType board_state_;
154 TranspositionTable<KeyType> transposition_table_;
155
156 void _ImplementUpdateBoardState(ExecutedMove move) {
157 // moving piece moves away from space
158 board_state_ = board_state_ ^ zkeys_.GetHashValue(
159 move.moving_piece.piece_color,
160 move.moving_piece.piece_type,
161 move.spaces.start
162 );
163
164 // if capture piece, remove from board
165 if (move.destination_piece.piece_color != PieceColor::kNul) {
166 board_state_ = board_state_ ^ zkeys_.GetHashValue(
167 move.destination_piece.piece_color,
168 move.destination_piece.piece_type,
169 move.spaces.end
170 );
171 }
172
173 // moving piece to new space
174 board_state_ = board_state_ ^ zkeys_.GetHashValue(
175 move.moving_piece.piece_color,
176 move.moving_piece.piece_type,
177 move.spaces.end
178 );
179
180 // change state now that its other player's turn
181 board_state_ = board_state_ ^ zkeys_.turn_key;
182 }
183};
