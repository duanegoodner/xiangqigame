83GamePointsSMap_t BPOPointsSKeys::ToGamePointsSmap() {
84 GamePointsSMap_t s_map{};
85
86 for (auto piece : black_base_) {
87 s_map["black"][piece.first] = utility_functs::array_plus_const(
88 black_position_[piece.first],
89 black_base_[piece.first]
90 );
91 }
92
93 for (auto piece : red_base_offsets_) {
94 auto red_base = black_base_[piece.first] + red_base_offsets_[piece.first];
95 auto red_position = utility_functs::two_array_sum(
96 black_position_[piece.first],
97 red_position_offsets_[piece.first]
98 );
99 s_map["red"][piece.first] = utility_functs::array_plus_const(red_position, red_base);
100 }
101
102 return s_map;
103}

188TeamPointsArray_t BPOPointsEKeys::RedNetPoints() {
189 TeamPointsArray_t red_net_points{};
190 for (auto piece : red_base_offsets_) {
191 auto base_points = black_base_[piece.first] + red_base_offsets_[piece.first];
192
193 auto unflipped_position_points = utility_functs::two_array_sum(
194 black_position_[piece.first],
195 red_position_offsets_[piece.first]
196 );
197
198 auto flipped_position_points =
199 utility_functs::vertical_flip_array(unflipped_position_points);
200
201 red_net_points[piece.first] =
202 utility_functs::array_plus_const(flipped_position_points, base_points);
203 }
204 return red_net_points;
205}
