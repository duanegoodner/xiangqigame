43 MoveCollection &team_moves
44) {
45
46 auto fwd_space = space + FwdDirection(color);

48 if (ExistsAndPassesColorTest(board_map, fwd_space, color)) {
49 team_moves.Append(Move{space, fwd_space});

62void PieceMoves::CannonMoves(

66 MoveCollection &team_moves
67) {
68 for (auto direction : kAllOrthogonalDirections) {
69 auto next_step = space + direction;
70 while (next_step.IsOnBoard() && (not is_occupied(board_map, next_step))) {
71 team_moves.Append(Move{space, next_step});

162void PieceMoves::FlyingGeneralMove(
163 const BoardMap_t &board_map,
164 PieceColor color,
165 const BoardSpace &space,
166 MoveCollection &team_moves
167) {
168
169 auto has_flying_move = true;
170 auto opponent_color = opponent_of(color);
171 auto other_gen_position = get_general_position(board_map, opponent_color);
172 if (space.file != other_gen_position.file) {
173 has_flying_move = false;
174 } else {
175 auto search_start = min(space.rank, other_gen_position.rank) + 1;
176 auto search_end = max(space.rank, other_gen_position.rank) - 1;
177 for (auto rank = search_start; rank <= search_end; rank++) {
178 if (is_occupied(board_map, BoardSpace{rank, space.file})) {
179 has_flying_move = false;
180 }
181 }
182 }
183
184 if (has_flying_move) {
185 team_moves.Append(Move{space, other_gen_position});
186 }
187}
