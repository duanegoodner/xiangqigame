46GamePointsSMap_t BPOPointsSKeys::ToGamePointsSmap() {
47 GamePointsSMap_t s_map{};
48
49 for (auto piece : black_base_) {
50 s_map["black"][piece.first] = utility_functs::array_plus_const(
51 black_position_[piece.first],
52 black_base_[piece.first]
53 );
54 }
55
56 for (auto piece : red_base_offsets_) {
57 auto red_base = black_base_[piece.first] + red_base_offsets_[piece.first];
58 auto red_position = utility_functs::two_array_sum(
59 black_position_[piece.first],
60 red_position_offsets_[piece.first]
61 );
62 s_map["red"][piece.first] = utility_functs::array_plus_const(red_position, red_base);
63 }
64
65 return s_map;
66}

151TeamPointsArray_t BPOPointsEKeys::RedNetPoints() {
152 TeamPointsArray_t red_net_points{};
153 for (auto piece : red_base_offsets_) {
154 auto base_points = black_base_[piece.first] + red_base_offsets_[piece.first];
155
156 auto unflipped_position_points = utility_functs::two_array_sum(
157 black_position_[piece.first],
158 red_position_offsets_[piece.first]
159 );
160
161 auto flipped_position_points =
162 utility_functs::vertical_flip_array(unflipped_position_points);
163
164 red_net_points[piece.first] =
165 utility_functs::array_plus_const(flipped_position_points, base_points);
166 }
167 return red_net_points;
168}
