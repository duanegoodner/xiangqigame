56 bool IsOnBoard() const {

60 bool IsInHomelandOf(const PieceColor color) const {
61 assert(color != PieceColor::kNul);
62 if (color == PieceColor::kRed) {
63 return rank >= kRedRiverEdge;
64 } else {
65 return rank <= kBlackRiverEdge;
66 }
67 }
68
69 bool IsInCastleOf(const PieceColor color) const {
70 assert(color != PieceColor::kNul);
71 auto castle_edges = color == PieceColor::kRed ? kRedCastleEdges : kBlackCastleEdges;
72 return (castle_edges.min_rank <= rank) && (rank <= castle_edges.max_rank) &&
73 (castle_edges.min_file <= file) && (file <= castle_edges.max_file);
74 }

92// Generates 1-D array of castle spaces from castle edge definitions
93inline constexpr Castle_t calc_castle_spaces(const CastleEdges &edges) {
94 Castle_t spaces{};
95
96 for (auto rank = edges.min_rank; rank <= edges.max_rank; rank++) {
97 for (auto file = edges.min_file; file <= edges.max_file; file++) {
98 spaces[3 * (rank - edges.min_rank) + file - edges.min_file].rank = rank;
99 spaces[3 * (rank - edges.min_rank) + file - edges.min_file].file = file;
100 }
101 }
102 return spaces;
103}

133inline BoardSpace get_general_position(
134 const BoardMap_t &board_map,
135 const PieceColor color
136) {
137 auto castle =
138 (color == PieceColor::kRed) ? red_castle_spaces() : black_castle_spaces();
139
140 BoardSpace found_space;
141
142 for (BoardSpace board_space : castle) {
143 auto piece = board_map[board_space.rank][board_space.file];
144 if (piece.piece_type == PieceType::kGen) {
145 found_space = board_space;
146 }
147 }
148 return found_space;
149}

151inline vector<BoardSpace> get_all_spaces_occupied_by(
152 const BoardMap_t &board_map,
153 const PieceColor color
154) {
155 vector<BoardSpace> occupied_spaces;
156 occupied_spaces.reserve(16);
157 for (auto rank = 0; rank < kNumRanks; rank++) {
158 for (auto file = 0; file < kNumFiles; file++) {
159 if (get_color(board_map, BoardSpace{rank, file}) == color) {
160 occupied_spaces.emplace_back(BoardSpace{rank, file});
161 }
162 }
163 }
164 return occupied_spaces;
