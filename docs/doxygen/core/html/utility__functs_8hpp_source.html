117template <typename T>
118bool hasRepeatingPattern(
119 const std::vector<T> &vec,
120 int lookback_length,
121 int period
122) {
123 // Ensure lookback_length is a multiple of period and the vector has at least
124 // lookback_length elements
125 if (lookback_length % period != 0 || vec.size() < lookback_length) {
126 return false;
127 }
128
129 // Get the number of repetitions
130 int repetitions = lookback_length / period;
131
132 // Check if the pattern with period repeats for the last lookback_length
133 // elements
134 for (int i = 0; i < period; ++i) {
135 T patternElement = vec[vec.size() - lookback_length + i];
136 for (int j = 1; j < repetitions; ++j) {
137 if (vec[vec.size() - lookback_length + i + j * period] !=
138 patternElement) {
139 return false;
140 }
141 }
142 }
143
144 return true;
145}
