58MoveCollection GameBoard::ImplementCalcFinalMovesOf(PieceColor color) {
59 auto un_tested_moves = move_calculator_.CalcAllMovesNoCheckTest(color, board_map_);
60 MoveCollection validated_moves;
61 validated_moves.moves.reserve(un_tested_moves.moves.size());
62
63 for (auto move : un_tested_moves.moves) {
64 auto executed_move = ImplementExecuteMove(move);
65 auto resulting_opponent_moves =
66 move_calculator_.CalcAllMovesNoCheckTest(opponent_of(color), board_map_);
67 auto resulting_gen_position = get_general_position(board_map_, color);
68
69 if (not resulting_opponent_moves.ContainsDestination(resulting_gen_position) and
70 not ViolatesRepeatRule(color)) {
71 validated_moves.Append(move);
72 }
73
74 ImplementUndoMove(executed_move);
75 }
76 return validated_moves;
77}

86ExecutedMove GameBoard::ImplementExecuteMove(Move move) {
87 auto moving_piece = GetOccupant(move.start);
88 auto destination_piece = GetOccupant(move.end);
89 SetOccupant(move.end, moving_piece);
90 SetOccupant(move.start, GamePiece(PieceType::kNnn, PieceColor::kNul));
91
92 auto executed_move = ExecutedMove{move, moving_piece, destination_piece};
93 UpdateHashCalculator(executed_move);
94 AddToMoveLog(executed_move);
95
96 return ExecutedMove{move, moving_piece, destination_piece};
97};

135void GameBoard::RemoveFromMoveLog(ExecutedMove executed_move) {
136 auto piece_color = executed_move.moving_piece.piece_color;
137 auto last_move_by_color = move_log_[piece_color].back();
138 if (!(executed_move == last_move_by_color)) {
139 throw runtime_error("Last move in log does not match move to be removed");
140 }
141 move_log_[piece_color].pop_back();
142}

144bool GameBoard::ViolatesRepeatRule(PieceColor color) {
145 for (auto period_length : kRepeatPeriodsToCheck) {
146 auto lookback_length = (kMaxAllowedRepeatPeriods + 1) * period_length;
147 if (utility_functs::hasRepeatingPattern(
148 move_log_[color],
149 lookback_length,
150 period_length
151 )) {
152 return true;
153 }
154 }
155 return false;
156}
