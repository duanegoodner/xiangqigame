59MoveCollection GameBoard::ImplementCalcFinalMovesOf(PieceColor color) {
60 auto un_tested_moves = move_calculator_.CalcAllMovesNoCheckTest(color, board_map_);
61 MoveCollection validated_moves;
62 validated_moves.moves.reserve(un_tested_moves.moves.size());
63
64 for (auto move : un_tested_moves.moves) {
65 auto executed_move = ImplementExecuteMove(move);
66 auto resulting_opponent_moves =
67 move_calculator_.CalcAllMovesNoCheckTest(opponent_of(color), board_map_);
68 auto resulting_gen_position = get_general_position(board_map_, color);
69
70 if (not resulting_opponent_moves.ContainsDestination(resulting_gen_position) and
71 not ViolatesRepeatRule(color)) {
72 validated_moves.Append(move);
73 }
74
75 ImplementUndoMove(executed_move);
76 }
77 return validated_moves;
78}

87ExecutedMove GameBoard::ImplementExecuteMove(Move move) {
88 auto moving_piece = GetOccupant(move.start);
89 auto destination_piece = GetOccupant(move.end);
90 SetOccupant(move.end, moving_piece);
91 SetOccupant(move.start, GamePiece(PieceType::kNnn, PieceColor::kNul));
92
93 auto executed_move = ExecutedMove{move, moving_piece, destination_piece};
94 UpdateHashCalculator(executed_move);
95 AddToMoveLog(executed_move);
96
97 return ExecutedMove{move, moving_piece, destination_piece};
98};

134void GameBoard::RemoveFromMoveLog(ExecutedMove executed_move) {
135 auto piece_color = executed_move.moving_piece.piece_color;
136 auto last_move_by_color = move_log_[piece_color].back();
137 if (!(executed_move == last_move_by_color)) {
138 throw runtime_error("Last move in log does not match move to be removed");
139 }
140 move_log_[piece_color].pop_back();
141}

143bool GameBoard::ViolatesRepeatRule(PieceColor color) {
144 for (auto period_length : kRepeatPeriodsToCheck) {
145 auto lookback_length = (kRepeatPeriodsMaxAllowed + 1) * period_length;
146 if (utility_functs::hasRepeatingPattern(
147 move_log_[color],
148 lookback_length,
149 period_length
150 )) {
151 return true;
152 }
153 }
154 return false;
155}
