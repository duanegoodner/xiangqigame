74 def format_board_output(self, board: GameBoard):
75 file_labels = [
76 f" {chr(char_num)} " for char_num in range(ord("a"), ord("j"))
77 ]
78 file_labels.insert(0, "\t")
79 file_labels.insert(len(file_labels), "\n")
80
81 board_list = [
82 [
83 f" {self.encode_piece(board.map()[row][col])} "
84 for col in range(len(board.map()[0]))
85 ]
86 for row in range(len(board.map()))
87 ]
88 for row_index in range(len(board_list)):
89 board_list[row_index].insert(0, f" {str(10 - row_index)}\t")
90 board_list.insert(0, file_labels)
91 board_list = ["".join(row) for row in board_list]
92
93 return str("\n\n".join([str(rank) for rank in board_list]))
