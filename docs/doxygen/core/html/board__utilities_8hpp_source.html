24inline BoardMap_t int_board_to_game_pieces(const BoardMapInt_t int_board) {

52inline BoardSpace get_general_position(
53 const BoardMap_t &board_map,
54 const PieceColor color
55) {
56 auto castle = (color == PieceColor::kRed) ? red_castle_spaces()
57 : black_castle_spaces();
58
59 BoardSpace found_space;
60
61 for (BoardSpace board_space : castle) {
62 auto piece = board_map[board_space.rank][board_space.file];
63 if (piece.piece_type == PieceType::kGen) {
64 found_space = board_space;
65 }
66 }
67 return found_space;
68}

70inline vector<BoardSpace> get_all_spaces_occupied_by(
71 const BoardMap_t &board_map,
72 const PieceColor color
73) {
74 vector<BoardSpace> occupied_spaces;
75 occupied_spaces.reserve(16);
76 for (auto rank = 0; rank < kNumRanks; rank++) {
77 for (auto file = 0; file < kNumFiles; file++) {
78 if (get_color(board_map, BoardSpace{rank, file}) == color) {
79 occupied_spaces.emplace_back(BoardSpace{rank, file});
80 }
81 }
82 }
83 return occupied_spaces;
84}
