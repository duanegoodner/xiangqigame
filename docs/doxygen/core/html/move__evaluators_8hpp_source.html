206// CLASS TEMPLATE: RandomMoveEvaluator
207// IMPLEMENTS INTERFACE:
208// MoveEvaluator
209// USES:
210// ConcreteSpaceInfoProvider (e.g. GameBoard) that implements
211// SpaceInfoProvider.
212// Randomly chooses one of the legal moves available to evaluating_player_.
213template <typename ConcreteSpaceInfoProvider>
214class RandomMoveEvaluator
215 : public MoveEvaluator<RandomMoveEvaluator<ConcreteSpaceInfoProvider>> {
216public:
217 RandomMoveEvaluator(
218 PieceColor evaluating_player,
219 ConcreteSpaceInfoProvider &game_board
220 )
221 : evaluating_player_{evaluating_player}
222 , game_board_{game_board} {};
223 Move ImplementSelectMove() {
224 auto allowed_moves = game_board_.CalcFinalMovesOf(evaluating_player_);
225 auto selected_move_index =
226 utility_functs::random((size_t)0, allowed_moves.moves.size() - 1);
227 return allowed_moves.moves[selected_move_index];
228 };
229
230private:
231 PieceColor evaluating_player_;
232 ConcreteSpaceInfoProvider &game_board_;
233};
