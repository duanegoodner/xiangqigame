76struct BoardSpace {
77
78 BoardIdx_t rank, file;
79
80 bool IsOnBoard() const {
81 return ((0 <= rank) && (rank < kNumRanks) && (0 <= file) && (file < kNumFiles));
82 }
83
84 bool IsInHomelandOf(const PieceColor color) const {
85 assert(color != PieceColor::kNul);
86 if (color == PieceColor::kRed) {
87 return rank >= kRedRiverEdge;
88 } else {
89 return rank <= kBlackRiverEdge;
90 }
91 }
92
93 bool IsInCastleOf(const PieceColor color) const {
94 assert(color != PieceColor::kNul);
95 auto castle_edges = color == PieceColor::kRed ? kRedCastleEdges : kBlackCastleEdges;
96 return (castle_edges.min_rank <= rank) && (rank <= castle_edges.max_rank) &&
97 (castle_edges.min_file <= file) && (file <= castle_edges.max_file);
98 }
99
100 BoardSpace operator+(const BoardDirection direction) const {
101 return BoardSpace{rank + direction.rank, file + direction.file};
102 }
103
104 bool operator==(const BoardSpace other) const {
105 return (rank == other.rank) && (file == other.file);
106 }
107
108 bool operator!=(const BoardSpace other) const {
109 return (rank != other.rank) || (file != other.file);
110 }
111};

113// Data struct for internal tracking of team's castle spaces as 1-D array, and
114// then
115typedef array<BoardSpace, 9> Castle_t;
116
117// Generates 1-D array of castle spaces from castle edge definitions
118inline constexpr Castle_t calc_castle_spaces(const CastleEdges &edges) {
119 Castle_t spaces{};
120
121 for (auto rank = edges.min_rank; rank <= edges.max_rank; rank++) {
122 for (auto file = edges.min_file; file <= edges.max_file; file++) {
123 spaces[3 * (rank - edges.min_rank) + file - edges.min_file].rank = rank;
124 spaces[3 * (rank - edges.min_rank) + file - edges.min_file].file = file;
125 }
126 }
127 return spaces;
128}

130constexpr Castle_t red_castle_spaces() { return calc_castle_spaces(kRedCastleEdges); }
