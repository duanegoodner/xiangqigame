293 @property
294 def largest_magnitude_noninf_eval_score(self) -> int:
295 largest_score_magnitudes = []
296 inf_like_values = [np.iinfo(PointsT).max, np.iinfo(PointsT).min]
297 for df in self.dfs.values():
298 if df is not None:
299 assert df["eval_score"].dtype == PointsT
300 non_inf_like_values = df[
301 ~df["eval_score"].isin(inf_like_values)
302 ]["eval_score"]
303 largest_score_magnitudes.append(
304 np.abs(non_inf_like_values.min())
305 )
306 largest_score_magnitudes.append(
307 np.abs(non_inf_like_values.max())
