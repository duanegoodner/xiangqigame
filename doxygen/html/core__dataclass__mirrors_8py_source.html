17class PointsTypeDeterminer:
18 """
19 Determines the integer type used for Points int the C++ core, and provides
20 a numpy integer type with the same size and signed-ness.
21 """
22
23 _dtype_map = {
24 (True, 1): np.int8,
25 (True, 2): np.int16,
26 (True, 4): np.int32,
27 (True, 8): np.int64,
28 (False, 1): np.uint8,
29 (False, 2): np.uint16,
30 (False, 4): np.uint32,
31 (False, 8): np.uint64,
32 }
33
34 def get_points_type(self) -> type:
35 dtype_key = (
36 bindings.is_signed_points_type(),
37 bindings.size_of_points_type(),
38 )
39 return self._dtype_map[dtype_key]
40
41
42PointsT: TypeAlias = PointsTypeDeterminer().get_points_type()
