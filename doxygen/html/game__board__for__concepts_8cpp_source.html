57MoveCollection GameBoardForConcepts::CalcFinalMovesOf(PieceColor color) {
58 auto un_tested_moves = move_calculator_.CalcAllMovesNoCheckTest(color, board_map_);
59 MoveCollection validated_moves{};
60
61 if (IsDraw()) {
62 return validated_moves;
63 }
64
65 validated_moves.moves.reserve(un_tested_moves.moves.size());
66
67 for (auto move : un_tested_moves.moves) {
68 auto executed_move = ExecuteMove(move);
69 auto resulting_opponent_moves =
70 move_calculator_.CalcAllMovesNoCheckTest(opponent_of(color), board_map_);
71 auto resulting_gen_position = get_general_position(board_map_, color);
72
73 if (not resulting_opponent_moves.ContainsDestination(resulting_gen_position) and
74 not ViolatesRepeatRule(color)) {
75 validated_moves.Append(move);
76 }
77
78 UndoMove(executed_move);
79 }
80 return validated_moves;
81}

90ExecutedMove GameBoardForConcepts::ExecuteMove(const Move &move) {
91 auto moving_piece = GetOccupantAt(move.start);
92 auto destination_piece = GetOccupantAt(move.end);
93 SetOccupantAt(move.end, moving_piece);
94 SetOccupantAt(move.start, GamePiece(PieceType::kNnn, PieceColor::kNul));
95
96 auto executed_move =
97 ExecutedMove{move, moving_piece, destination_piece, moves_since_last_capture_};
98 UpdateStateTracker(executed_move);
99 AddToMoveLog(executed_move);
100 if (!IsCaptureMove(executed_move)) {
101 moves_since_last_capture_++;
102 } else {
103 moves_since_last_capture_ = 0;
104 }
105
106 return executed_move;
107};

170bool GameBoardForConcepts::ViolatesRepeatRule(PieceColor color) {
