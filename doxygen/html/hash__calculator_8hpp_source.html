59  executed_move.moving_piece.piece_color,
60  executed_move.moving_piece.piece_type,
61  executed_move.spaces.start
62  );
63 
64  // if capture piece, remove from board
65  if (executed_move.destination_piece.piece_color != PieceColor::kNul) {
66  board_state_ = board_state_ ^ GetHashValueAt(
67  executed_move.destination_piece.piece_color,
68  executed_move.destination_piece.piece_type,
69  executed_move.spaces.end
70  );
71  }

75  executed_move.moving_piece.piece_color,
76  executed_move.moving_piece.piece_type,
77  executed_move.spaces.end
78  );
79 
80  // change state now that its other player's turn
