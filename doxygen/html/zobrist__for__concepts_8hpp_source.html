164class TranspositionTableForConcepts {
165 std::unordered_map<KeyType, TranspositionTableEntryForConcepts<KeyType, NumConfKeys>>
166 data_;
167 MoveCountType move_counter_;
168
169public:
170 TranspositionTableForConcepts() = default;
171
172 moveselection::TranspositionTableSearchResult GetDataAt(
173 KeyType primary_board_state,
174 DepthType remaining_search_depth,
175 std::array<KeyType, NumConfKeys> expected_keys
176 ) {
177 moveselection::TranspositionTableSearchResult result{};
178 auto tr_table_entry_it = data_.find(primary_board_state);
179 if (tr_table_entry_it != data_.end()) {
180 auto tr_table_entry = tr_table_entry_it->second;
181 if (tr_table_entry.remaining_search_depth() >= remaining_search_depth) {
182 tr_table_entry.set_last_access_index(move_counter_);
183 result.set_found(true);
184 result.set_minimax_calc_result(tr_table_entry.minimax_calc_result());
185 }
186 if (result.found() and
187 !tr_table_entry.ConfirmationKeysMatchExpected(expected_keys)) {
188 result.set_known_collision(true);
189 }
190 }
191 return result;
192 }
193

195 KeyType primary_board_state,
196 DepthType search_depth,
197 moveselection::MinimaxResultType result_type,
198 moveselection::EqualScoreMoves &similar_moves,
199 const std::array<KeyType, NumConfKeys> &confirmation_keys
200 ) {
201 data_.insert_or_assign(
202 primary_board_state,
203 TranspositionTableEntryForConcepts<KeyType, NumConfKeys>{
204 moveselection::MinimaxCalcResult{search_depth, result_type, similar_moves},
205 confirmation_keys,
206 move_counter_
207 }
208 );
209 }
