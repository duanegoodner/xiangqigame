93 board_state_ = board_state_ ^ GetHashValueAt(

105 executed_move.moving_piece.piece_color,
106 executed_move.moving_piece.piece_type,
107 executed_move.spaces.start
108 );
109
110 // if capture piece, remove from board
111 if (executed_move.destination_piece.piece_color != gameboard::PieceColor::kNul) {
112 board_state_ = board_state_ ^ GetHashValueAt(
113 executed_move.destination_piece.piece_color,
114 executed_move.destination_piece.piece_type,
115 executed_move.spaces.end
116 );
117 }

120 board_state_ = board_state_ ^ GetHashValueAt(
121 executed_move.moving_piece.piece_color,
122 executed_move.moving_piece.piece_type,
123 executed_move.spaces.end
124 );
125
126 // change state now that its other player's turn
127 board_state_ = board_state_ ^ turn_key();

282 DepthType remaining_search_depth,
283 std::array<KeyType, NumConfKeys> expected_keys
284 ) {
285 moveselection::TranspositionTableSearchResult result{};
286 auto tr_table_entry_it = data_.find(primary_board_state);
287 if (tr_table_entry_it != data_.end()) {
288 auto tr_table_entry = tr_table_entry_it->second;

290 tr_table_entry.set_last_access_index(move_counter_);

293 }
294 if (result.found() and

296 result.set_known_collision(true);
297 }
298 }

300 }
