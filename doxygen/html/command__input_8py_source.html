37class PlayerCommandInterpreter:
38 """
39 Converts command line input related to a player into PlayerInput object.
40 """
41
42 _player_input_dispatch = {
43 "ai": PlayerType.AI,
44 "person": PlayerType.HUMAN,
45 None: PlayerType.AI,
46 }
47
48 _minimax_key_size_dispatch = {
49 32: MinimaxMoveEvaluator32,
50 64: MinimaxMoveEvaluator64,
51 128: MinimaxMoveEvaluator128,
52 }
53
54 _default_key_size = 64
55
56 _default_strength = 4
57
58 _default_number_zobrist_states = 1
59
60 def __init__(
61 self,
62 player_input: str,
63 algo_input: str,
64 strength_input: int,
65 key_size_input: int,
66 zkeys_seed: int,
67 number_zobrist_states: int,
68 ):
69 self.player_input = player_input
70 self.algo_input = algo_input
71 self.strength_input = strength_input
72 self.key_size_input = key_size_input
73 self.zkeys_seed = zkeys_seed
74 self.number_zobrist_states = number_zobrist_states
75
76 def _get_key_size(self) -> int:
77 if self.key_size_input is None:
78 return self._default_key_size
79 else:
80 return self.key_size_input
81
82 def _get_strength(self) -> int:
83 if self.strength_input is None:
84 return self._default_strength
85 else:
86 return self.strength_input
87
88 def _get_number_zobrist_states(self) -> int:
89 if self.number_zobrist_states is None:
90 return self._default_number_zobrist_states
91 else:
92 return self.number_zobrist_states
93
94 def interpret_command(self) -> PlayerInput:
95 player_type = self._player_input_dispatch[self.player_input]
96 if player_type == PlayerType.HUMAN:
97 algo = strength = key_size = num_zobrist_states = None
98 else:
99 if self.algo_input == "random":
100 algo = EvaluatorType.RANDOM
101 strength = key_size = num_zobrist_states = None
102 else:
103 key_size = self._get_key_size()
104 algo = EvaluatorType.MINIMAX
105 strength = self._get_strength()
106 num_zobrist_states = self._get_number_zobrist_states()
107 return PlayerInput(
108 player_type=player_type,
109 algo=algo,
110 strength=strength,
111 key_size=key_size,
112 zkeys_seed=self.zkeys_seed,
113 num_zobrist_states=num_zobrist_states,
114 )
