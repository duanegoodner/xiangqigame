321 @property
322 def largest_magnitude_noninf_eval_score(self) -> int:
323 largest_score_magnitudes = []
324 inf_like_values = [np.iinfo(PointsT).max, np.iinfo(PointsT).min]
325 for df in self.dfs.values():
326 if df is not None:
327 assert df["eval_score"].dtype == PointsT
328 non_inf_like_values = df[
329 ~df["eval_score"].isin(inf_like_values)
330 ]["eval_score"]
331 largest_score_magnitudes.append(
332 np.abs(non_inf_like_values.min())
333 )
334 largest_score_magnitudes.append(
335 np.abs(non_inf_like_values.max())
336 )
337 return max(largest_score_magnitudes)

339 @staticmethod
340 def symmetric_winsorize(
341 data: pd.Series, magnitude: int | float
342 ) -> pd.Series:
343 assert magnitude >= 0
344 return cast(
345 pd.Series, np.clip(data, a_min=-1 * magnitude, a_max=magnitude)
