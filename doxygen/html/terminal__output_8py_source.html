87 def format_board_output(self, board: GameBoard):
88 file_labels = [
89 f" {chr(char_num)} " for char_num in range(ord("a"), ord("j"))
90 ]
91 file_labels.insert(0, "\t")
92 file_labels.insert(len(file_labels), "\n")
93
94 board_list = [
95 [
96 f" {self.encode_piece(board.map()[row][col])} "
97 for col in range(len(board.map()[0]))
98 ]
99 for row in range(len(board.map()))
100 ]
101 for row_index in range(len(board_list)):
102 board_list[row_index].insert(0, f" {str(10 - row_index)}\t")
103 board_list.insert(0, file_labels)
104 board_list = ["".join(row) for row in board_list]
105
106 return str("\n\n".join([str(rank) for rank in board_list]))
