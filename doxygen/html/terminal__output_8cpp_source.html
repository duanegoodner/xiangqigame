120std::string BoardMapEncoder::EncodeBoardMap(const gameboard::BoardMap_t &board_map) {
121 std::ostringstream board_output;
122
123 board_output << "\t";
124 for (char file = 'a'; file <= 'i'; ++file) {
125 board_output << " " << file << " ";
126 }
127 board_output << "\n\n\n";
128 // Board rows
129 for (size_t row = 0; row < board_map.size(); ++row) {
130 board_output << " " << 10 - row << "\t"; // Row labels (10 to 1)
131
132 for (const auto &piece : board_map[row]) {
133 board_output << " " << game_piece_encoder_.EncodeGamePiece(piece) << " ";
134 }
135
136 board_output << "\n\n";
137 }
138
139 return board_output.str();
140}
