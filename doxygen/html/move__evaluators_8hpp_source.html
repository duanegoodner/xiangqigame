154 ScoredMove RateMove(Move move, PieceColor cur_player) {
155 auto piece_type = game_board_.GetType(move.start);
156
157 auto end_points = game_position_points_
158 .GetValueOfPieceAtPosition(cur_player, piece_type, move.end);
159 auto start_points = game_position_points_.GetValueOfPieceAtPosition(
160 cur_player,
161 piece_type,
162 move.start
163 );
164 auto position_value_delta = end_points - start_points;
165
166 Points_t capture_val;
167
168 if (game_board_.GetColor(move.end) == opponent_of(cur_player)) {
169 auto captured_piece_type = game_board_.GetType(move.end);
170 capture_val = GetValueOfPieceAtPosition(
171 opponent_of(cur_player),
172 captured_piece_type,
173 move.end
174 );
175 } else {
176 capture_val = 0;
177 }
178
179 return ScoredMove{move, (position_value_delta + capture_val)};
180 }

190 std::vector<ScoredMove> GenerateRankedMoveList(

193 ) {
194 vector<ScoredMove> rated_moves;

349 EqualScoreMoves FinalizeNodeResult(
350 MinimaxResultType &result_type,
351 int best_eval,
352 MoveCollection best_moves,
353 int remaining_search_depth,
354 SearchSummary &search_summary
355 ) {
356
357 // If result_type is still unknown by the time FinalizeNodeResult is called, then we
358 // have evaluated all moves from (i.e. children of) the node. Can achieve this for
359 // root node or interior node, but not for a leaf node.
360 if (result_type == MinimaxResultType::kUnknown) {

410 int RecursivelyVisitNodes(
411 Move move,
412 PieceColor cur_player,
413 MoveCollection &allowed_moves,
414 int remaining_search_depth,
415 int alpha,
416 int beta,
417 SearchSummary &search_summary,
418 bool use_transposition_table
419 ) {
420 auto executed_move = game_board_.ExecuteMove(move);
421 auto new_allowed_moves = game_board_.CalcFinalMovesOf(opponent_of(cur_player));
422 auto cur_eval = MinimaxRec(
423 new_allowed_moves,
424 remaining_search_depth - 1,
425 alpha,
426 beta,
427 opponent_of(cur_player),
428 search_summary,
429 use_transposition_table
430 )
431 .shared_score;
432 game_board_.UndoMove(executed_move);
433 return cur_eval;
434 }

452
453 EqualScoreMoves HandleInternalNode(
454 PieceColor cur_player,
455 MoveCollection &allowed_moves,
456 int &remaining_search_depth,
457 int &alpha,
458 int &beta,
459 MinimaxResultType result_type,
460 SearchSummary &search_summary,
461 bool use_transposition_table
462 ) {
463 auto max_eval = InitializedBestEval(cur_player);
464 MoveCollection best_moves;
465 auto ranked_moves = GenerateRankedMoveList(cur_player, allowed_moves);
466
467 for (auto rated_move : ranked_moves) {
468 auto cur_eval = RecursivelyVisitNodes(
469 rated_move.move,
470 cur_player,
471 allowed_moves,
472 remaining_search_depth,
473 alpha,
474 beta,
475 search_summary,
476 use_transposition_table
477 );
478
479 UpdateBestMoves(cur_player, rated_move.move, best_moves, cur_eval, max_eval);
480
481 UpdatePruningParam(alpha, beta, cur_eval, cur_player);
482
483 if (IsPrunable(alpha, beta, result_type, cur_player)) {
484 break;
485 }
486 }
487
488 return FinalizeNodeResult(
489 result_type,
490 max_eval,
491 best_moves,
492 remaining_search_depth,
493 search_summary
494 );
495 }
496
497 EqualScoreMoves MinimaxRec(
498 MoveCollection &allowed_moves,
499 int remaining_search_depth,
500 int alpha,
501 int beta,
502 PieceColor cur_player,
503 SearchSummary &search_summary,
504 bool use_transposition_table
505 ) {
506 MinimaxResultType result_type{};
507
508 // Check if result for current board state is in transposition table
509 // (unless this is a second search in which case we don't use transposition table)
510 if (use_transposition_table) {
511 auto tr_table_search_result = hash_calculator_.GetTrData(remaining_search_depth);
512 if (tr_table_search_result.found &&
513 tr_table_search_result.IsConsistentWith(allowed_moves)) {
514 return HandleTrTableHit(
515 search_summary,
516 result_type,
517 tr_table_search_result,
518 remaining_search_depth
519 );
520 }
521 }
522
523 // If no legal moves, node is an end-of-game leaf
524 if (allowed_moves.moves.size() == 0) {
525 return HandleEndOfGame(
526 cur_player,
527 search_summary,
528 result_type,
529 remaining_search_depth
530 );
531 }
532
533 // If remaining search depth is zero, treat node as "normal" leaf (not end of game,
534 // just end of search)
535 if (remaining_search_depth == 0) {
536 return HandleLeaf(cur_player, search_summary, result_type, remaining_search_depth);
537 }
538
539 // If not end-of-game, nor end-of-search, treat node as internal node.
540 return HandleInternalNode(
541 cur_player,
542 allowed_moves,
543 remaining_search_depth,
544 alpha,
545 beta,
546 result_type,
547 search_summary,
548 use_transposition_table
549 );
550 }

552 void RunMinimax(
553 MoveCollection &allowed_moves,
554 SearchSummary &search_summary,
555 bool use_transposition_table = true
556 ) {
557
558 auto search_start = std::chrono::high_resolution_clock::now();
559 auto minimax_result = MinimaxRec(
560 allowed_moves,
561 starting_search_depth_,
562 numeric_limits<Points_t>::min(),
563 numeric_limits<Points_t>::max(),
564 evaluating_player_,
565 search_summary,
566 use_transposition_table
567 );
568 auto search_end = std::chrono::high_resolution_clock::now();
569 std::chrono::duration<double, std::nano> search_time = search_end - search_start;
570 search_summary.set_time(search_time);
571 auto selected_move_index =
572 utility_functs::random((size_t)0, minimax_result.similar_moves.moves.size() - 1);
573 auto selected_move = minimax_result.similar_moves.moves[selected_move_index];
574 search_summary.SetSelectedMove(selected_move);
575 auto tr_table_size = hash_calculator_.GetTrTableSize();
576 search_summary.set_tr_table_size_final(tr_table_size);
