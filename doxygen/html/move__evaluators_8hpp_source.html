67template <typename ConcreteSpaceInfoProvider, typename ConcretePieceValueProvider>
68class PreSearchMoveSorter {
69public:
70 PreSearchMoveSorter(
71 ConcreteSpaceInfoProvider &game_board,
72 ConcretePieceValueProvider &game_position_points
73 )
74 : game_board_{game_board}
75 , game_position_points_{game_position_points} {}
76
77 ScoredMove RateMove(Move move, PieceColor cur_player) {
78 auto piece_type = game_board_.GetType(move.start);
79
80 auto end_points = game_position_points_
81 .GetValueOfPieceAtPosition(cur_player, piece_type, move.end);
82 auto start_points = game_position_points_.GetValueOfPieceAtPosition(
83 cur_player,
84 piece_type,
85 move.start
86 );
87 auto position_value_delta = end_points - start_points;
88
89 Points_t capture_val;
90
91 if (game_board_.GetColor(move.end) == opponent_of(cur_player)) {
92 auto captured_piece_type = game_board_.GetType(move.end);
93 capture_val = game_position_points_.GetValueOfPieceAtPosition(
94 opponent_of(cur_player),
95 captured_piece_type,
96 move.end
97 );
98 } else {
99 capture_val = 0;
100 }
101
102 return ScoredMove{move, (position_value_delta + capture_val)};
103 }
104
105 std::vector<ScoredMove> GenerateRankedMoveList(
106 PieceColor cur_player,
107 const MoveCollection &cur_player_moves
108 ) {
109 vector<ScoredMove> rated_moves;
110 for (auto cur_move : cur_player_moves.moves) {
111 auto cur_rated_move = RateMove(cur_move, cur_player);
112 rated_moves.emplace_back(cur_rated_move);
113 }
114 sort(
115 rated_moves.begin(),
116 rated_moves.end(),
117 [](const ScoredMove &move_a, const ScoredMove &move_b) {
118 return move_a.score > move_b.score;
119 }
120 );
121 return rated_moves;
122 }
123
124private:
125 ConcreteSpaceInfoProvider &game_board_;
126 ConcretePieceValueProvider &game_position_points_;
127};

419 Points_t RecursivelyVisitNodes(
420 Move move,
421 PieceColor cur_player,
422 const MoveCollection &allowed_moves,
423 DepthType search_depth,
424 Points_t alpha,
425 Points_t beta,
426 SearchSummary &search_summary,
427 bool use_transposition_table
428 ) {
429 auto executed_move = game_board_.ExecuteMove(move);
430 auto new_allowed_moves = game_board_.CalcFinalMovesOf(opponent_of(cur_player));
431 auto cur_eval = MinimaxRecursive(
432 new_allowed_moves,
433 search_depth - 1,
434 alpha,
435 beta,
436 opponent_of(cur_player),
437 search_summary,
438 use_transposition_table
439 )
440 .shared_score;
441 game_board_.UndoMove(executed_move);
442 return cur_eval;
443 }

466 EqualScoreMoves HandleInternalNode(
467 PieceColor cur_player,
468 const MoveCollection &allowed_moves,
469 DepthType &search_depth,
470 Points_t &alpha,
471 Points_t &beta,
472 MinimaxResultType result_type,
473 SearchSummary &search_summary,
474 bool use_transposition_table
475 ) {
476 auto max_eval = InitializedBestEval(cur_player);
477 MoveCollection best_moves;
478 auto ranked_moves = move_sorter_.GenerateRankedMoveList(cur_player, allowed_moves);
479
480 for (auto rated_move : ranked_moves) {
481 auto cur_eval = RecursivelyVisitNodes(
482 rated_move.move,
483 cur_player,
484 allowed_moves,
485 search_depth,
486 alpha,
487 beta,
488 search_summary,
489 use_transposition_table
490 );
491
492 UpdateBestMoves(cur_player, rated_move.move, best_moves, cur_eval, max_eval);
493
494 UpdatePruningParam(alpha, beta, cur_eval, cur_player);
495
496 if (IsPrunable(alpha, beta, result_type, cur_player)) {
497 break;
498 }
499 }
500
501 return FinalizeNodeResult(
502 result_type,
503 max_eval,
504 best_moves,
505 search_depth,
506 search_summary
507 );
508 }

510 EqualScoreMoves MinimaxRecursive(
511 const MoveCollection &allowed_moves,
512 DepthType search_depth,
513 Points_t alpha,
514 Points_t beta,
515 PieceColor cur_player,
516 SearchSummary &search_summary,
517 bool use_transposition_table
518 ) {
519 MinimaxResultType result_type{};
520
521 // Check if result for current board state is in transposition table
522 // (unless this is a second search in which case we don't use transposition table)
523 if (use_transposition_table) {
524 auto tr_table_search_result =
525 hash_calculator_.GetTrData(search_depth, num_move_selections_);
526 if (tr_table_search_result.found() &&
527 tr_table_search_result.IsConsistentWith(allowed_moves)) {
528 return HandleTrTableHit(
529 search_summary,
530 result_type,
531 tr_table_search_result,
532 search_depth
533 );
534 }
535 }
536
537 // If no legal moves, node is an end-of-game leaf
538 if (allowed_moves.moves.size() == 0) {
539 return HandleEndOfGame(cur_player, search_summary, result_type, search_depth);
540 }
541
542 // If remaining search depth is zero, treat node as "normal" leaf (not end of game,
543 // just end of search)
544 if (search_depth == 0) {
545 return HandleLeaf(cur_player, search_summary, result_type, search_depth);
546 }
547
548 // If not end-of-game, nor end-of-search, treat node as internal node.
549 return HandleInternalNode(
550 cur_player,
551 allowed_moves,
552 search_depth,
553 alpha,
554 beta,
555 result_type,
556 search_summary,
557 use_transposition_table
558 );
559 }

561 EqualScoreMoves RunTimedMinimax(
562 const MoveCollection &allowed_moves,
563 SearchSummary &search_summary,
564 bool use_transposition_table = true
565 ) {
566 auto search_start = std::chrono::high_resolution_clock::now();
567 auto minimax_result = MinimaxRecursive(
568 allowed_moves,
569 search_depth_,
570 numeric_limits<Points_t>::min(),
571 numeric_limits<Points_t>::max(),
572 evaluating_player_,
573 search_summary,
574 use_transposition_table
575 );
576 auto search_end = std::chrono::high_resolution_clock::now();
577 std::chrono::duration<double, std::nano> search_time = search_end - search_start;
578 search_summary.set_time(search_time);
579
580 return minimax_result;
581 }

766 EqualScoreMoves EvaluateNonWinLeaf(
767 PieceColor cur_player,
768 MinimaxResultType &result_type
769 ) {
770 result_type = MinimaxResultType::kStandardLeaf;
771
772 auto cur_player_points = GetPlayerTotal(cur_player);
773 auto opponent_points = GetPlayerTotal(opponent_of(cur_player));
774
775 auto empty_move_collection = MoveCollection();
776
777 if (cur_player == evaluating_player_) {
778 return EqualScoreMoves{
779 (cur_player_points - opponent_points),
780 empty_move_collection
781 };
782 } else {
783 return EqualScoreMoves{
784 (opponent_points - cur_player_points),
785 empty_move_collection
786 };
787 }
788 };

799 search_summary.RecordNodeInfo(result_type, search_depth, result);

807 PieceColor cur_player
808 ) {

816 void UpdateBestMoves(
817 PieceColor cur_player,
818 Move move,

822 ) {

824 best_moves.Append(move);
825 } else if (IsImprovement(cur_eval, previous_best_eval, cur_player)) {

832 EqualScoreMoves FinalizeNodeResult(
833 MinimaxResultType &result_type,
834 Points_t best_eval,
835 MoveCollection best_moves,
836 DepthType search_depth,
837 SearchSummary &search_summary
838 ) {
839
840 // If result_type is still unknown by the time FinalizeNodeResult is called, then we
841 // have evaluated all moves from (i.e. children of) the node. Can achieve this for
842 // root node or interior node, but not for a leaf node.
843 if (result_type == MinimaxResultType::kUnknown) {
844 result_type = MinimaxResultType::kFullyEvaluatedNode;
845 }
846 EqualScoreMoves result{best_eval, best_moves};
847 hash_calculator_
848 .RecordTrData(search_depth, result_type, result, num_move_selections_);
849 search_summary.RecordNodeInfo(result_type, search_depth, result);
850 return result;
851 }

866 Points_t RecursivelyVisitNodes(
867 Move move,
868 PieceColor cur_player,
869 const MoveCollection &allowed_moves,
870 DepthType search_depth,
871 Points_t alpha,
872 Points_t beta,
873 SearchSummary &search_summary,
874 bool use_transposition_table
875 ) {
876 auto executed_move = game_board_.ExecuteMove(move);
877 auto new_allowed_moves = game_board_.CalcFinalMovesOf(opponent_of(cur_player));
878 auto cur_eval = MinimaxRecursive(
879 new_allowed_moves,
880 search_depth - 1,
881 alpha,
882 beta,
883 opponent_of(cur_player),
884 search_summary,
885 use_transposition_table
886 )
887 .shared_score;
888 game_board_.UndoMove(executed_move);
889 return cur_eval;
890 }

901 Points_t &alpha,
902 Points_t &beta,
903 Points_t cur_eval,
904 PieceColor cur_player
905 ) {

907 minimaxutils::UpdateAlpha(alpha, cur_eval);

912
913 EqualScoreMoves HandleInternalNode(
914 PieceColor cur_player,
915 const MoveCollection &allowed_moves,
916 DepthType &search_depth,
917 Points_t &alpha,
918 Points_t &beta,
919 MinimaxResultType result_type,
920 SearchSummary &search_summary,
921 bool use_transposition_table
922 ) {
923 auto max_eval = InitializedBestEval(cur_player);
924 MoveCollection best_moves;
925 auto ranked_moves = move_sorter_.GenerateRankedMoveList(cur_player, allowed_moves);
926
927 for (auto rated_move : ranked_moves) {
928 auto cur_eval = RecursivelyVisitNodes(
929 rated_move.move,
930 cur_player,
931 allowed_moves,
932 search_depth,
933 alpha,
934 beta,
935 search_summary,
936 use_transposition_table
937 );
938
939 UpdateBestMoves(cur_player, rated_move.move, best_moves, cur_eval, max_eval);
940
941 UpdatePruningParam(alpha, beta, cur_eval, cur_player);
942
943 if (IsPrunable(alpha, beta, result_type, cur_player)) {
944 break;
945 }
946 }
947
948 return FinalizeNodeResult(
949 result_type,
950 max_eval,
951 best_moves,
952 search_depth,
953 search_summary
954 );
955 }
956
957 EqualScoreMoves MinimaxRecursive(
958 const MoveCollection &allowed_moves,
959 DepthType search_depth,
960 Points_t alpha,
961 Points_t beta,
962 PieceColor cur_player,
963 SearchSummary &search_summary,
964 bool use_transposition_table
965 ) {
966 MinimaxResultType result_type{};
967
968 // Check if result for current board state is in transposition table
969 // (unless this is a second search in which case we don't use transposition table)
970 if (use_transposition_table) {
971 auto tr_table_search_result =
972 hash_calculator_.GetTrData(search_depth, num_move_selections_);
973 if (tr_table_search_result.found() &&
974 tr_table_search_result.IsConsistentWith(allowed_moves)) {
975 return HandleTrTableHit(
976 search_summary,
977 result_type,
978 tr_table_search_result,
979 search_depth
980 );
981 }
982 }
983
984 // If no legal moves, node is an end-of-game leaf
985 if (allowed_moves.moves.size() == 0) {
986 return HandleEndOfGame(cur_player, search_summary, result_type, search_depth);
987 }
988
989 // If remaining search depth is zero, treat node as "normal" leaf (not end of game,
990 // just end of search)
991 if (search_depth == 0) {
992 return HandleLeaf(cur_player, search_summary, result_type, search_depth);
993 }
994
995 // If not end-of-game, nor end-of-search, treat node as internal node.
996 return HandleInternalNode(
997 cur_player,
998 allowed_moves,
999 search_depth,
1000 alpha,
1001 beta,
1002 result_type,
1003 search_summary,
1004 use_transposition_table
1005 );
1006 }

1008 EqualScoreMoves RunTimedMinimax(
1009 const MoveCollection &allowed_moves,
1010 SearchSummary &search_summary,
1011 bool use_transposition_table = true
1012 ) {
1013 auto search_start = std::chrono::high_resolution_clock::now();
1014 auto minimax_result = MinimaxRecursive(
1015 allowed_moves,
1016 search_depth_,
1017 numeric_limits<Points_t>::min(),
1018 numeric_limits<Points_t>::max(),
1019 evaluating_player_,
1020 search_summary,
1021 use_transposition_table
1022 );
1023 auto search_end = std::chrono::high_resolution_clock::now();
1024 std::chrono::duration<double, std::nano> search_time = search_end - search_start;
1025 search_summary.set_time(search_time);
1026
1027 return minimax_result;
1028 }
