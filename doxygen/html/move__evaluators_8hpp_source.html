58template <typename ConcreteSpaceInfoProvider, typename ConcretePieceValueProvider>
59class PreSearchMoveSorter {
60public:
61 PreSearchMoveSorter(
62 ConcreteSpaceInfoProvider &game_board,
63 ConcretePieceValueProvider &game_position_points
64 )
65 : game_board_{game_board}
66 , game_position_points_{game_position_points} {}
67
68 ScoredMove RateMove(Move move, PieceColor cur_player) {
69 auto piece_type = game_board_.GetType(move.start);
70
71 auto end_points = game_position_points_
72 .GetValueOfPieceAtPosition(cur_player, piece_type, move.end);
73 auto start_points = game_position_points_.GetValueOfPieceAtPosition(
74 cur_player,
75 piece_type,
76 move.start
77 );
78 auto position_value_delta = end_points - start_points;
79
80 Points_t capture_val;
81
82 if (game_board_.GetColor(move.end) == opponent_of(cur_player)) {
83 auto captured_piece_type = game_board_.GetType(move.end);
84 capture_val = game_position_points_.GetValueOfPieceAtPosition(
85 opponent_of(cur_player),
86 captured_piece_type,
87 move.end
88 );
89 } else {
90 capture_val = 0;
91 }
92
93 return ScoredMove{move, (position_value_delta + capture_val)};
94 }
95
96 std::vector<ScoredMove> GenerateRankedMoveList(
97 PieceColor cur_player,
98 const MoveCollection &cur_player_moves
99 ) {
100 vector<ScoredMove> rated_moves;
101 for (auto cur_move : cur_player_moves.moves) {
102 auto cur_rated_move = RateMove(cur_move, cur_player);
103 rated_moves.emplace_back(cur_rated_move);
104 }
105 sort(
106 rated_moves.begin(),
107 rated_moves.end(),
108 [](const ScoredMove &move_a, const ScoredMove &move_b) {
109 return move_a.score > move_b.score;
110 }
111 );
112 return rated_moves;
113 }
114
115private:
116 ConcreteSpaceInfoProvider &game_board_;
117 ConcretePieceValueProvider &game_position_points_;
118};

389 Points_t RecursivelyVisitNodes(
390 Move move,
391 PieceColor cur_player,
392 const MoveCollection &allowed_moves,
393 DepthType remaining_search_depth,
394 Points_t alpha,
395 Points_t beta,
396 SearchSummary &search_summary,
397 bool use_transposition_table
398 ) {
399 auto executed_move = game_board_.ExecuteMove(move);
400 auto new_allowed_moves = game_board_.CalcFinalMovesOf(opponent_of(cur_player));
401 auto cur_eval = MinimaxRec(
402 new_allowed_moves,
403 remaining_search_depth - 1,
404 alpha,
405 beta,
406 opponent_of(cur_player),
407 search_summary,
408 use_transposition_table
409 )
410 .shared_score;
411 game_board_.UndoMove(executed_move);
412 return cur_eval;
413 }

436 EqualScoreMoves HandleInternalNode(
437 PieceColor cur_player,
438 const MoveCollection &allowed_moves,
439 DepthType &remaining_search_depth,
440 Points_t &alpha,
441 Points_t &beta,
442 MinimaxResultType result_type,
443 SearchSummary &search_summary,
444 bool use_transposition_table
445 ) {
446 auto max_eval = InitializedBestEval(cur_player);
447 MoveCollection best_moves;
448 auto ranked_moves = move_sorter_.GenerateRankedMoveList(cur_player, allowed_moves);
449
450 for (auto rated_move : ranked_moves) {
451 auto cur_eval = RecursivelyVisitNodes(
452 rated_move.move,
453 cur_player,
454 allowed_moves,
455 remaining_search_depth,
456 alpha,
457 beta,
458 search_summary,
459 use_transposition_table
460 );
461
462 UpdateBestMoves(cur_player, rated_move.move, best_moves, cur_eval, max_eval);
463
464 UpdatePruningParam(alpha, beta, cur_eval, cur_player);
465
466 if (IsPrunable(alpha, beta, result_type, cur_player)) {
467 break;
468 }
469 }
470
471 return FinalizeNodeResult(
472 result_type,
473 max_eval,
474 best_moves,
475 remaining_search_depth,
476 search_summary
477 );
478 }
479
480 EqualScoreMoves MinimaxRec(
481 const MoveCollection &allowed_moves,
482 DepthType remaining_search_depth,
483 Points_t alpha,
484 Points_t beta,
485 PieceColor cur_player,
486 SearchSummary &search_summary,
487 bool use_transposition_table
488 ) {
489 MinimaxResultType result_type{};
490
491 // Check if result for current board state is in transposition table
492 // (unless this is a second search in which case we don't use transposition table)
493 if (use_transposition_table) {
494 auto tr_table_search_result =
495 hash_calculator_.GetTrData(remaining_search_depth, num_move_selections_);
496 if (tr_table_search_result.found() &&
497 tr_table_search_result.IsConsistentWith(allowed_moves)) {
498 return HandleTrTableHit(
499 search_summary,
500 result_type,
501 tr_table_search_result,
502 remaining_search_depth
503 );
504 }
505 }
506
507 // If no legal moves, node is an end-of-game leaf
508 if (allowed_moves.moves.size() == 0) {
509 return HandleEndOfGame(
510 cur_player,
511 search_summary,
512 result_type,
513 remaining_search_depth
514 );
515 }
516
517 // If remaining search depth is zero, treat node as "normal" leaf (not end of game,
518 // just end of search)
519 if (remaining_search_depth == 0) {
520 return HandleLeaf(cur_player, search_summary, result_type, remaining_search_depth);
521 }
522
523 // If not end-of-game, nor end-of-search, treat node as internal node.
524 return HandleInternalNode(
525 cur_player,
526 allowed_moves,
527 remaining_search_depth,
528 alpha,
529 beta,
530 result_type,
531 search_summary,
532 use_transposition_table
533 );
534 }

536 EqualScoreMoves RunTimedMinimax(
537 const MoveCollection &allowed_moves,
538 SearchSummary &search_summary,
539 bool use_transposition_table = true
540 ) {
541 auto search_start = std::chrono::high_resolution_clock::now();
542 auto minimax_result = MinimaxRec(
543 allowed_moves,
544 starting_search_depth_,
545 numeric_limits<Points_t>::min(),
546 numeric_limits<Points_t>::max(),
547 evaluating_player_,
548 search_summary,
549 use_transposition_table
550 );
551 auto search_end = std::chrono::high_resolution_clock::now();
552 std::chrono::duration<double, std::nano> search_time = search_end - search_start;
553 search_summary.set_time(search_time);
554
555 return minimax_result;
556 }
