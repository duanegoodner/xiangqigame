59 // Calculation methods
60

69 KeyType GetHashValueAt(PieceColor color, PieceType piece_type, BoardSpace space) {
70 return zarray_[GetZColorIndexOf(color)][piece_type][space.rank][space.file];
71 }
72
73private:
75 static const GameZarray_t CreateGameZarray(
76 PseudoRandomKeyGenerator<KeyType> &key_generator
77 ) {
78 GameZarray_t game_zarray{};
79 for (auto color_idx = 0; color_idx < 2; color_idx++) {
80 for (auto piece_id = 1; piece_id < kNumPieceTypeVals; piece_id++) {
81 for (auto rank = 0; rank < kNumRanks; rank++) {
82 for (auto file = 0; file < kNumFiles; file++) {
83 game_zarray[color_idx][piece_id][rank][file] = key_generator.GenerateKey();
84 }
85 }
86 }
87 }
88 return game_zarray;
89 }
90
91 // Calculation methods
92
93 void FullBoardStateCalInternal(const gameboard::BoardMap_t &board_map) {
94 board_state_ = 0;
95 for (size_t rank = 0; rank < gameboard::kNumRanks; rank++) {
96 for (size_t file = 0; file < gameboard::kNumFiles; file++) {
97 if (board_map[rank][file].piece_color != 0) {
98 board_state_ = board_state_ ^ GetHashValueAt(
99 board_map[rank][file].piece_color,
100 board_map[rank][file].piece_type,
101 gameboard::BoardSpace{(int)rank, (int)file}
102 );
103 }
104 }
105 }
106 }
107
108 void UpdateBoardStateInternal(const gameboard::ExecutedMove &executed_move) {
109 board_state_ = board_state_ ^ GetHashValueAt(
110 executed_move.moving_piece.piece_color,
111 executed_move.moving_piece.piece_type,
112 executed_move.spaces.start
113 );
114
115 // if capture piece, remove from board
116 if (executed_move.destination_piece.piece_color != gameboard::PieceColor::kNul) {
117 board_state_ = board_state_ ^ GetHashValueAt(
118 executed_move.destination_piece.piece_color,
119 executed_move.destination_piece.piece_type,
120 executed_move.spaces.end
121 );
122 }
123
124 // moving piece to new space
125 board_state_ = board_state_ ^ GetHashValueAt(
126 executed_move.moving_piece.piece_color,
127 executed_move.moving_piece.piece_type,
128 executed_move.spaces.end
129 );
130
131 // change state now that its other player's turn
132 board_state_ = board_state_ ^ turn_key();
133 }

276template <typename KeyType, size_t NumConfKeys>
277class TranspositionTable {
278 std::unordered_map<KeyType, TranspositionTableEntry<KeyType, NumConfKeys>> data_;
279 MoveCountType move_counter_;
280
281public:
282 TranspositionTable() = default;
283
284 moveselection::TranspositionTableSearchResult GetDataAt(
285 KeyType primary_board_state,
286 DepthType remaining_search_depth,
287 std::array<KeyType, NumConfKeys> expected_keys
288 ) {
289 moveselection::TranspositionTableSearchResult result{};
290 auto tr_table_entry_it = data_.find(primary_board_state);
291 if (tr_table_entry_it != data_.end()) {
292 auto tr_table_entry = tr_table_entry_it->second;
293 if (tr_table_entry.remaining_search_depth() >= remaining_search_depth) {
294 tr_table_entry.set_last_access_index(move_counter_);
295 result.set_found(true);
296 result.set_minimax_calc_result(tr_table_entry.minimax_calc_result());
297 }
298 if (result.found() and
299 !tr_table_entry.ConfirmationKeysMatchExpected(expected_keys)) {
300 result.set_known_collision(true);
301 }
302 }
303 return result;
304 }
305
306 void RecordData(
307 KeyType primary_board_state,
308 DepthType search_depth,
309 moveselection::MinimaxResultType result_type,
310 moveselection::EqualScoreMoves &similar_moves,
311 const std::array<KeyType, NumConfKeys> &confirmation_keys
312 ) {
313 data_.insert_or_assign(
314 primary_board_state,
315 TranspositionTableEntry<KeyType, NumConfKeys>{
316 moveselection::MinimaxCalcResult{search_depth, result_type, similar_moves},
317 confirmation_keys,
318 move_counter_
319 }
320 );
321 }
322
323 size_t size() { return data_.size(); }
324
325 void IncrementMoveCounter() { move_counter_++; }
326
327 MoveCountType NumIdleMovesAt(KeyType board_state) {
328 auto tr_table_entry_it = data_.at(board_state);
329 return NumMovesSinceLastUseOf(tr_table_entry_it->second);
330 }
331
332private:
333 MoveCountType NumMovesSinceLastUseOf(
334 const TranspositionTableEntry<KeyType, NumConfKeys> &tr_table_entry
335 ) {
336 return move_counter_ - tr_table_entry.last_access_index();
337 }
