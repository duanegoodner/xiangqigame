31 void FullBoardStateCalc(const BoardMap_t &board_map) {
32 board_state_ = 0;
33 for (size_t rank = 0; rank < kNumRanks; rank++) {
34 for (size_t file = 0; file < kNumFiles; file++) {
35 if (board_map[rank][file].piece_color != 0) {
36 board_state_ = board_state_ ^ GetHashValueAt(
37 board_map[rank][file].piece_color,
38 board_map[rank][file].piece_type,
39 BoardSpace{(int)rank, (int)file}
40 );
41 }
42 }
43 }
44 }

46 void UpdateBoardState(const ExecutedMove &executed_move) {
47 board_state_ = board_state_ ^ GetHashValueAt(
48 executed_move.moving_piece.piece_color,
49 executed_move.moving_piece.piece_type,
50 executed_move.spaces.start
51 );
52
53 // if capture piece, remove from board
54 if (executed_move.destination_piece.piece_color != PieceColor::kNul) {
55 board_state_ = board_state_ ^ GetHashValueAt(
56 executed_move.destination_piece.piece_color,
57 executed_move.destination_piece.piece_type,
58 executed_move.spaces.end
59 );
60 }
61
62 // moving piece to new space
63 board_state_ = board_state_ ^ GetHashValueAt(
64 executed_move.moving_piece.piece_color,
65 executed_move.moving_piece.piece_type,
66 executed_move.spaces.end
67 );
68
69 // change state now that its other player's turn
70 board_state_ = board_state_ ^ turn_key();
71 }

106 static const GameZarray_t CreateGameZarray(
107 PseudoRandomKeyGenerator<KeyType> &key_generator
108 ) {
109 GameZarray_t game_zarray{};
110 for (auto color_idx = 0; color_idx < 2; color_idx++) {
111 for (auto piece_id = 1; piece_id < kNumPieceTypeVals; piece_id++) {
112 for (auto rank = 0; rank < kNumRanks; rank++) {
113 for (auto file = 0; file < kNumFiles; file++) {
114 game_zarray[color_idx][piece_id][rank][file] = key_generator.GenerateKey();
115 }
116 }
117 }
