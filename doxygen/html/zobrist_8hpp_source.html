21template <typename KeyType>
22class ZobristCalculator {
23public:
24 typedef array<array<KeyType, gameboard::kNumFiles>, gameboard::kNumRanks>
25 PieceZarray_t;
26 typedef array<PieceZarray_t, gameboard::kNumPieceTypeVals> TeamZarray_t;
27 typedef array<TeamZarray_t, 2> GameZarray_t;
28
29 ZobristCalculator(uint32_t seed)
30 : zarray_{}
31 , turn_key_{}
32 , board_state_{}
33 , seed_{seed} {
34 PseudoRandomKeyGenerator<KeyType> key_generator{seed};
35 turn_key_ = key_generator.GenerateKey();
36 zarray_ = CreateGameZarray(key_generator);
37 };
38
39 ZobristCalculator()
40 : ZobristCalculator(random_device{}()) {}
41
42 void FullBoardStateCalc(const gameboard::BoardMap_t &board_map) {
43 board_state_ = 0;
44 for (size_t rank = 0; rank < gameboard::kNumRanks; rank++) {
45 for (size_t file = 0; file < gameboard::kNumFiles; file++) {
46 if (board_map[rank][file].piece_color != 0) {
47 board_state_ = board_state_ ^ GetHashValueAt(
48 board_map[rank][file].piece_color,
49 board_map[rank][file].piece_type,
50 gameboard::BoardSpace{(int)rank, (int)file}
51 );
52 }
53 }
54 }
55 }
56
57 void UpdateBoardState(const gameboard::ExecutedMove &executed_move) {
58 board_state_ = board_state_ ^ GetHashValueAt(
59 executed_move.moving_piece.piece_color,
60 executed_move.moving_piece.piece_type,
61 executed_move.spaces.start
62 );
63
64 // if capture piece, remove from board
65 if (executed_move.destination_piece.piece_color != gameboard::PieceColor::kNul) {
66 board_state_ = board_state_ ^ GetHashValueAt(
67 executed_move.destination_piece.piece_color,
68 executed_move.destination_piece.piece_type,
69 executed_move.spaces.end
70 );
71 }
72
73 // moving piece to new space
74 board_state_ = board_state_ ^ GetHashValueAt(
75 executed_move.moving_piece.piece_color,
76 executed_move.moving_piece.piece_type,
77 executed_move.spaces.end
78 );
79
80 // change state now that its other player's turn
81 board_state_ = board_state_ ^ turn_key();
82 }
83
84 KeyType GetHashValueAt(PieceColor color, PieceType piece_type, BoardSpace space) {
85 return zarray_[GetZColorIndexOf(color)][piece_type][space.rank][space.file];
86 }
87
88 KeyType board_state() const { return board_state_; }
89 GameZarray_t zarray() const { return zarray_; }
90 KeyType turn_key() const { return turn_key_; }
91 uint32_t seed() const { return seed_; }
92
93private:
94 GameZarray_t zarray_;
95 KeyType turn_key_;
96 uint32_t seed_;
97 KeyType board_state_;
98
99 static const GameZarray_t CreateGameZarray(
100 PseudoRandomKeyGenerator<KeyType> &key_generator
101 ) {
102 GameZarray_t game_zarray{};
103 for (auto color_idx = 0; color_idx < 2; color_idx++) {
104 for (auto piece_id = 1; piece_id < kNumPieceTypeVals; piece_id++) {
105 for (auto rank = 0; rank < kNumRanks; rank++) {
106 for (auto file = 0; file < kNumFiles; file++) {
107 game_zarray[color_idx][piece_id][rank][file] = key_generator.GenerateKey();
108 }
109 }
110 }
111 }
112 return game_zarray;
113 };

246template <typename KeyType, size_t NumConfKeys>
247class TranspositionTable {
248public:
249 moveselection::TranspositionTableSearchResult GetDataAt(
250 KeyType primary_board_state,
251 DepthType remaining_search_depth,
252 std::array<KeyType, NumConfKeys> expected_keys
253 // MoveCountType access_index
254 ) {
255 moveselection::TranspositionTableSearchResult result{};
256 auto tr_table_entry_it = data_.find(primary_board_state);
257 if (tr_table_entry_it != data_.end()) {
258 auto tr_table_entry = tr_table_entry_it->second;
259 if (tr_table_entry.remaining_search_depth() >= remaining_search_depth) {
260 tr_table_entry.set_last_access_index(move_counter_);
261 result.set_found(true);
262 result.set_minimax_calc_result(tr_table_entry.minimax_calc_result());
263 }
264 if (result.found() and
265 !tr_table_entry.ConfirmationKeysMatchExpected(expected_keys)) {
266 result.set_known_collision(true);
267 }
268 }
269 return result;
270 }
271
272 void RecordData(
273 KeyType primary_board_state,
274 DepthType search_depth,
275 moveselection::MinimaxResultType result_type,
276 moveselection::EqualScoreMoves &similar_moves,
277 const std::array<KeyType, NumConfKeys> &confirmation_keys
278 // MoveCountType access_index
279 ) {
280 data_.insert_or_assign(
281 primary_board_state,
282 TranspositionTableEntry<KeyType, NumConfKeys>{
283 moveselection::MinimaxCalcResult{search_depth, result_type, similar_moves},
284 confirmation_keys,
285 move_counter_
286 }
287 );
288 }
289
290 size_t size() { return data_.size(); }
291
292 void IncrementMoveCounter() { move_counter_++; }
293
294private:
295 std::unordered_map<KeyType, TranspositionTableEntry<KeyType, NumConfKeys>> data_;
296 MoveCountType move_counter_;
297
298 MoveCountType NumMovesSinceLastUseOf(
299 const TranspositionTableEntry<KeyType, NumConfKeys> &tr_table_entry
300 ) {
301 return move_counter_ - tr_table_entry.last_access_index();
302 }
