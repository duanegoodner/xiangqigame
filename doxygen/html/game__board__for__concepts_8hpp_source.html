169 MoveCollection CalcFinalMovesOfInternal(PieceColor color) {
170 auto un_tested_moves = move_calculator_.CalcAllMovesNoCheckTest(color, board_map_);
171 MoveCollection validated_moves{};
172
173 if (IsDraw()) {
174 return validated_moves;
175 }
176
177 validated_moves.moves.reserve(un_tested_moves.moves.size());
178
179 for (auto move : un_tested_moves.moves) {
180 auto executed_move = ExecuteMove(move);
181 auto resulting_opponent_moves =
182 move_calculator_.CalcAllMovesNoCheckTest(opponent_of(color), board_map_);
183 auto resulting_gen_position = get_general_position(board_map_, color);
184
185 if (not resulting_opponent_moves.ContainsDestination(resulting_gen_position) and
186 not ViolatesRepeatRule(color)) {
187 validated_moves.Append(move);
188 }
189
190 UndoMove(executed_move);
191 }
192 return validated_moves;
193 }

202 ExecutedMove ExecuteMoveInternal(const Move &move) {
203 auto moving_piece = GetOccupantAt(move.start);
204 auto destination_piece = GetOccupantAt(move.end);
205 SetOccupantAt(move.end, moving_piece);
206 SetOccupantAt(move.start, GamePiece(PieceType::kNnn, PieceColor::kNul));
207
208 auto executed_move =
209 ExecutedMove{move, moving_piece, destination_piece, moves_since_last_capture_};
210 UpdateStateTracker(executed_move);
211 AddToMoveLog(executed_move);
212 if (!IsCaptureMove(executed_move)) {
213 moves_since_last_capture_++;
214 } else {
215 moves_since_last_capture_ = 0;
216 }
217
218 return executed_move;
219 }

252 void RemoveFromMoveLog(const ExecutedMove &executed_move) {
253 auto piece_color = executed_move.moving_piece.piece_color;
254 auto last_move_by_color = move_log_[piece_color].back();
255 if (!(executed_move == last_move_by_color)) {
256 throw runtime_error("Last move in log does not match move to be removed");
257 }
258 move_log_[piece_color].pop_back();
259 }

261 bool ViolatesRepeatRule(PieceColor color) {
262 for (auto period_length : kRepeatPeriodsToCheck) {
263 auto lookback_length = (kRepeatPeriodsMaxAllowed + 1) * period_length;
264 if (utility_functs::hasRepeatingPattern(
265 move_log_[color],
266 lookback_length,
267 period_length
268 )) {
269 return true;
270 }
271 }
272 return false;
273 }
