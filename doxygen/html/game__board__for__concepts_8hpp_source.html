123 ExecutedMove ExecuteMoveInternal(const Move &move) {
124 auto moving_piece = GetOccupantAt(move.start);
125 auto destination_piece = GetOccupantAt(move.end);
126 SetOccupantAt(move.end, moving_piece);
127 SetOccupantAt(move.start, GamePiece(PieceType::kNnn, PieceColor::kNul));
128
129 auto executed_move =
130 ExecutedMove{move, moving_piece, destination_piece, moves_since_last_capture_};
131 UpdateStateTracker(executed_move);
132 AddToMoveLog(executed_move);
133 if (!IsCaptureMove(executed_move)) {
134 moves_since_last_capture_++;
135 } else {
136 moves_since_last_capture_ = 0;
137 }
138
139 return executed_move;
140 }

182 auto last_move_by_color = move_log_[piece_color].back();
183 if (!(executed_move == last_move_by_color)) {
184 throw runtime_error("Last move in log does not match move to be removed");
185 }
186 move_log_[piece_color].pop_back();
187 }
